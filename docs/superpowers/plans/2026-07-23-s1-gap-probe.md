  const gpRows = []; let gpDropped = 0;
  if (gpPresent) for (const l of section(probeT, 'Findings')) {
    if (!/^\s*\|/.test(l)) continue;
    const cells = l.split('|').slice(1, -1).map(c => c.trim());

    if (cells.every(c => /^:?-+:?$/.test(c))) continue; // separator row
